const int pinAO = 34;  // Entrada anal√≥gica
const int pinDO = 14;  // Entrada digital
const int buzzer = 27; // Buzzer activo

void setup() {
  Serial.begin(115200);
  delay(500);

  pinMode(pinDO, INPUT);
  pinMode(buzzer, OUTPUT);

  Serial.println("üî• MQ-2 + Buzzer Ready ‚Äî alarma modo speedrun activada");
}

void loop() {
  int valorAO = analogRead(pinAO);
  int valorDO = digitalRead(pinDO);

  Serial.print("AO: ");
  Serial.print(valorAO);
  Serial.print(" | DO: ");
  Serial.println(valorDO);

  // Alarma super r√°pida
  if (valorDO == HIGH || valorAO > 2000) {
    Serial.println("‚ö†Ô∏è GAS DETECTADO ‚Äî TURBO ALARMA üîîüî•");

    digitalWrite(buzzer, HIGH);
    delay(40);   // pitido muy corto
    digitalWrite(buzzer, LOW);
    delay(40);   // pausa muy corta

  } else {
    digitalWrite(buzzer, LOW);
  }

  delay(50);
}
